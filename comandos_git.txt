# El presente documento contiene todos los comandos que usas con git

# iniciar un nuevo proyecto
git init

# adicionar archivos a la memoria git, el "." indica subir todos los archivos del repositorio local
git add .

# grabar todos los cambios de la memoria git al repositorio master
git commit -m "comentario de la version subida, siempre se debe diligenciar"

# mostrar el historial de versiones
git log

# crear un nuevo branch
git checkout -b <nombre del nuevo branch>

# moverse entre branch
git chechout <nombre del branch hacia el que me muevo>

# listar branch del proyecto
git branch


# moverse entre ramas git
git checkout <nombre de la rama o branch>


# hacer merge entre branch,
# 1- ubicarme en el branch que va ha quedar como principal (master)
# 2- hacer git merge con el nombre de la rama que vamos a unir.
git merge file_new
